Html

JS

$('#expand-form').validation({
	validationcallBack: function(status, dom, errorMsg, defaultValue) {
		status || layer.tips(errorMsg, dom);
	},
	addOnRules: {
		zip: {
			validate: function(value) {
				return /^[1-9]\d{5}(?!\d)$/.test(value);
			},
			errorMsg: 'Input Zip code', 
			defaultValue: '400000' 
		}
	}
})

RUN
